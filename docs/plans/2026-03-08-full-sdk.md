```go
// types.go
package stt

// Response holds a basic transcription result (response_format=json).
type Response struct {
	Text     string  `json:"text"`
	Language string  `json:"language,omitempty"`
	Duration float64 `json:"duration,omitempty"`
}

// VerboseResponse holds a detailed transcription result (response_format=verbose_json).
type VerboseResponse struct {
	Text               string      `json:"text"`
	Language           string      `json:"language"`
	Duration           float64     `json:"duration"`
	Segments           []Segment   `json:"segments,omitempty"`
	Words              []Word      `json:"words,omitempty"`
	LanguageConfidence *float64    `json:"language_confidence,omitempty"`
	Utterances         []Utterance `json:"utterances,omitempty"`
}

// Segment is a time-aligned group of words.
type Segment struct {
	ID    int     `json:"id"`
	Start float64 `json:"start"`
	End   float64 `json:"end"`
	Text  string  `json:"text"`
}

// Word is a single word with timing and optional speaker.
type Word struct {
	Word    string `json:"word"`
	Start   float64 `json:"start"`
	End     float64 `json:"end"`
	Speaker *int   `json:"speaker,omitempty"`
}

// Utterance is a speaker-attributed segment (diarization).
type Utterance struct {
	Speaker int     `json:"speaker"`
	Start   float64 `json:"start"`
	End     float64 `json:"end"`
	Text    string  `json:"text"`
}

// Model describes an available STT model.
type Model struct {
	ID      string `json:"id"`
	Object  string `json:"object"`
	OwnedBy string `json:"owned_by"`
}

// ModelList is the response from GET /v1/models.
type ModelList struct {
	Object string  `json:"object"`
	Data   []Model `json:"data"`
}
```

```go
package stt

import "fmt"

// Error represents an STT API error with HTTP status code.
type Error struct {
	StatusCode int
	Message    string
}

func (e *Error) Error() string {
	return fmt.Sprintf("stt: HTTP %d: %s", e.StatusCode, e.Message)
}

// IsTransient returns true for errors that may succeed on retry (429, 502-504).
func (e *Error) IsTransient() bool {
	switch e.StatusCode {
	case 429, 502, 503, 504:

	}
	return false
}
```

### Task 3: Retry and Circuit Breaker

**Files:**

**Step 1: Create retry.go**

```go
package stt

import (
	"context"
	"errors"
	"math"
	"sync"
	"time"
)

type retryConfig struct {
	maxAttempts int
	baseDelay   time.Duration
	maxDelay    time.Duration
}

type circuitBreaker struct {
	mu         sync.Mutex
	failures   int
	maxFails   int
	cooldown   time.Duration
	openUntil  time.Time
}

func (cb *circuitBreaker) allow() bool {
	cb.mu.Lock()
	defer cb.mu.Unlock()
	if cb.failures >= cb.maxFails && time.Now().Before(cb.openUntil) {
		return false
	}
	if cb.failures >= cb.maxFails {
		cb.failures = 0 // half-open
	}
	return true
}

func (cb *circuitBreaker) recordSuccess() {
	cb.mu.Lock()
	cb.failures = 0
	cb.mu.Unlock()
}

func (cb *circuitBreaker) recordFailure() {
	cb.mu.Lock()
	cb.failures++
	if cb.failures >= cb.maxFails {
		cb.openUntil = time.Now().Add(cb.cooldown)
	}
	cb.mu.Unlock()
}

// doWithRetry wraps an operation with retry logic and circuit breaker.
func doWithRetry[T any](ctx context.Context, rc *retryConfig, cb *circuitBreaker, fn func() (T, error)) (T, error) {
	var zero T
	for attempt := range rc.maxAttempts {
		if cb != nil && !cb.allow() {
			return zero, &Error{StatusCode: 503, Message: "circuit breaker open"}
		}
		result, err := fn()
		if err == nil {
			if cb != nil { cb.recordSuccess() }
			return result, nil
		}
		var sttErr *Error
		if !errors.As(err, &sttErr) || !sttErr.IsTransient() {
			if cb != nil { cb.recordFailure() }
			return zero, err
		}
		if cb != nil { cb.recordFailure() }
		if attempt == rc.maxAttempts-1 {
			return zero, err
		}
		delay := time.Duration(float64(rc.baseDelay) * math.Pow(2, float64(attempt)))
		if delay > rc.maxDelay { delay = rc.maxDelay }
		select {
		case <-ctx.Done():
			return zero, ctx.Err()
		case <-time.After(delay):
		}
	}
	return zero, &Error{StatusCode: 503, Message: "max retries exceeded"}
}
```

```go
package stt

// StreamParams configures a WebSocket streaming session.
type StreamParams struct {
	Language      string // default: "en"
	VAD           bool   // default: true
	InterimResults bool  // default: false
	SmartFormat   bool
	Punctuate     bool   // default: true
	Encoding      string // "pcm_s16le" (default) or "pcm_f32le"
	SampleRate    int    // default: 16000
}

// StreamEvent is received from the server during streaming.
type StreamEvent struct {
	Type           string       `json:"type"` // Metadata, Results, SpeechStarted, Error, CloseStream
	RequestID      string       `json:"request_id,omitempty"`
	Model          string       `json:"model,omitempty"`
	Channels       int          `json:"channels,omitempty"`
	IsFinal        bool         `json:"is_final,omitempty"`
	SpeechFinal    bool         `json:"speech_final,omitempty"`
	FromFinalize   bool         `json:"from_finalize,omitempty"`
	SpeechStartedS *float64     `json:"speech_started_s,omitempty"`
	TimestampS     float64      `json:"timestamp_s,omitempty"`
	Channel        *StreamChannel `json:"channel,omitempty"`
	Message        string       `json:"message,omitempty"` // Error.message
}

// StreamChannel contains transcription alternatives.
type StreamChannel struct {
	Alternatives []StreamAlternative `json:"alternatives"`
}

// StreamAlternative is one transcription hypothesis.
type StreamAlternative struct {
	Transcript string  `json:"transcript"`
	Confidence float32 `json:"confidence"`
	Words      []Word  `json:"words,omitempty"`
}

// Transcript returns the best transcript from the first alternative, or "".
func (e *StreamEvent) Transcript() string {
	if e.Channel == nil || len(e.Channel.Alternatives) == 0 { return "" }
	return e.Channel.Alternatives[0].Transcript
}

// StreamHandler receives streaming events.
type StreamHandler interface {
	OnEvent(event StreamEvent)
	OnError(err error)
}

// StreamClient manages a WebSocket connection to /v1/listen.
type StreamClient struct {
	baseURL string
	params  StreamParams
	conn    *websocket.Conn
	handler StreamHandler
	done    chan struct{}
}
```

**Step 3: Implement StreamClient methods**

```go
package stt

import (
	"context"
	"fmt"
	"io"
	"net/http"
	"os"
	"path/filepath"
)

// TranscribeURL downloads audio from a URL, transcribes it, and cleans up.
// Useful for Telegram voice messages where you have the file download URL.
func (c *Client) TranscribeURL(ctx context.Context, audioURL string) (*Response, error) {
	tmp, err := downloadToTemp(ctx, audioURL)
	if err != nil {
		return nil, fmt.Errorf("download audio: %w", err)
	}
	defer os.Remove(tmp)
	return c.Transcribe(ctx, tmp)
}

// TranscribeURLVerbose is like TranscribeURL but returns verbose results.
func (c *Client) TranscribeURLVerbose(ctx context.Context, audioURL string) (*VerboseResponse, error) {
	tmp, err := downloadToTemp(ctx, audioURL)
	if err != nil {
		return nil, fmt.Errorf("download audio: %w", err)
	}
	defer os.Remove(tmp)
	return c.TranscribeVerbose(ctx, tmp)
}

func downloadToTemp(ctx context.Context, url string) (string, error) {
	req, err := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
	if err != nil {
		return "", err
	}
	resp, err := http.DefaultClient.Do(req)
	if err != nil {
		return "", err
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		return "", fmt.Errorf("HTTP %d", resp.StatusCode)
	}

	f, err := os.CreateTemp("", "stt-voice-*.ogg")
	if err != nil {
		return "", err
	}
	defer f.Close()

	if _, err := io.Copy(f, resp.Body); err != nil {
		os.Remove(f.Name())
		return "", err
	}
	return f.Name(), nil
}
```
